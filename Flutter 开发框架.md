- |--lib
    * |--common
		* |-- common.base
	* |--entry
	* |--model

	* |--res
	* |--router
	* |--test
	* |--ui

```
├── common - 一些通用的常量定义
├── model - 项目中使用的后台返回的数据模型
├── net - 基于dio的网络请求封装
│   └── interceptors - 网络请求的拦截器
├── pages  - 页面
│   ├── classify - 分类页面
│   ├── create - 资源创建编辑页面
│   ├── detail - 资源详情页面
│   ├── home - 首页
│   ├── login - 登录注册页面
│   ├── my_product_list - 我发布资源列表页面
│   ├── profile - 个人中心页面
│   └── web - web页面
├── route - 基于fluro的路由
├── utils - 项目中使用的工具类
├── vo - 前端页面组合的VO
└── widgets - 一些通用Widget
```
